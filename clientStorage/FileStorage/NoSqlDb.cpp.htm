#include "NoSqlDb.h"

  elem1.name = "elem1";
  elem1.category = "test";
  elem1.description = "elem1 description data";
  elem1.data = "elem1's StrData";

  elem3.data = "elem3's StrData";

  std::cout << "\n ------------------------------------------\n";

  std::cout << "\n  size of db = " << db.count() << "\n";
  Keys keys = db.keys();
  for (Key key : keys)
  {


    std::cout << "\n  " << key << ":";
    std::cout << db.value(key).show();
  }
